--Дата выезда в отдельную таблицу

insert into alone_time(date_start, fk_every_set_operation, fk_alone)
select date_exit, '3', key_alone
	from alone
	where date_exit is not null


--Дата категории в таблицу
insert into category_time(category, date_start, fk_alone)
select category, '01.01.2017', fk_alone
	from category
